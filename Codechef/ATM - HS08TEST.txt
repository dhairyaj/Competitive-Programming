#include <iostream>
#include <iomanip>
using namespace std;

int main() {
	long long int withdraw;
    long double balance;
    cin>>withdraw>>balance;

    if((withdraw+0.5<=balance)&&(withdraw%5==0))
        cout<<fixed<<setprecision(2)<<balance-withdraw-0.5;
    else
        cout<<fixed<<setprecision(2)<<balance;
    return 0;
}
