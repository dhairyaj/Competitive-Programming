#include <iostream>
using namespace std;

int main() {
	int t,n,c;
	cin>>t;
	for(int i = 0;i<t;i++){
	    cin>>n>>c;
	    int arr[n];
	    for(int j= 0;j<n;j++)
	        cin>>arr[j];
	    for(int j = 0;j<n;j++)
	        c -= arr[j];
	    if(c<0)
	        cout<<"No"<<endl;
	    else
	        cout<<"Yes"<<endl;
	}
	return 0;
}
